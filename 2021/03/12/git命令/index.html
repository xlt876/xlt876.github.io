git command

追加提交

git add .
git commit --amend

撤销提交

git reset HEAD

暂存

git stash

创建并切换分支

git checkout -b branch_name

切换分支

git checkout branch_name

查看分支

git branch

将A分支的某次提交‘合并(提交)’到分支B

git checkout B
git log  #get the commitID
git checkout A
git cherry-pick xxxxxx # xxxxxx refer the commitID

删除分支

git branch -d xxxx

远程

查看远程仓库

git remote -v

添加远程仓库

git remote add origin url/to/your/fork

本地仓库同步

git pull ssh://xlt@59.72.114.217:6888/home/xlt/mount/Xlt_Work/Isolate_RS/ares_parallel/src_p
git pull --rebase #追加提交远程库有更新需要merge时,不自动新增本地提交
git push --force-with-lease 509git #追加提交后push到远程仓库
