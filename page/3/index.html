awk 'xxxxxx; print "' $aa '"xxxxxx' #错误
awk 'xxxxxx; print "'$aa'"xxxxxx' #中间单引号与变量之间的空格会被shell当分隔符处理

awk 'xxxxxx; print "'” $aa “'"xxxxxx' #变量内有空格
